<head>
    <meta charset="UTF-8">
    <script src="https://telegram.org"></script>
    <style>
        body { font-family: sans-serif; background: var(--tg-theme-bg-color); color: var(--tg-theme-text-color); padding: 20px; }
        select, input { width: 100%; padding: 12px; margin: 10px 0; border-radius: 8px; border: 1px solid #ccc; box-sizing: border-box; }
        button { width: 100%; padding: 15px; background: var(--tg-theme-button-color); color: var(--tg-theme-button-text-color); border: none; border-radius: 8px; font-weight: bold; }
    </style>
</head>
<body>
    <h3>Отчет со стройплощадки</h3>
    <label>Объект:</label>
    <select id="project"></select>
    <input type="text" id="axis" placeholder="Ось / Этаж">
    <input type="number" id="volume" placeholder="Объем (м2)">
    <button onclick="send()">Отправить данные</button>

    <script>
        let tg = window.Telegram.WebApp;
        tg.expand();
        
        // Заполняем список из URL
        const params = new URLSearchParams(window.location.search);
        const projects = params.get('projects').split(',');
        const select = document.getElementById('project');
        projects.forEach(p => {
            let opt = document.createElement('option');
            opt.value = p; opt.innerHTML = p;
            select.appendChild(opt);
        });

        function send() {
            let data = {
                project: document.getElementById('project').value,
                axis: document.getElementById('axis').value,
                volume: document.getElementById('volume').value
            };
            tg.sendData(JSON.stringify(data));
            tg.close();
        }
    </script>
</body>
